# FakeC2
Install:

Termux:

1. pkg update && pkg upgrade
2. pkg install git
3. pkg install python3
4. git clone 
5. cd fakeC2
6. python3 fakeC2.py

kali-linux:

1. apt update && apt upgrade
2. apt install git
3. apt install python3
4. apt install python3-pip
5. git clone 
6. cd fakeC2
7. python3 fakeC2.py
